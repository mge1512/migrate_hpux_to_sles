%
% Better formatting of code blocks
%

% 1. Load tcolorbox and fvextra
\usepackage[most]{tcolorbox}
\usepackage{fvextra}

% 2. Define colors
\definecolor{code-bg}{HTML}{F5F7F9}
\definecolor{code-frame}{HTML}{E1E4E8}

% 3. The Smart Definition
% We check if 'Shaded' is defined. If yes, we renew it. If no, we create it.
\ifdefined\Shaded
  \renewenvironment{Shaded}
  {\begin{tcolorbox}[enhanced, breakable, colback=code-bg, colframe=code-frame, boxrule=0.5pt, arc=3mm, shadow={0mm}{0mm}{0mm}{white}]}
  {\end{tcolorbox}}
\else
  \newenvironment{Shaded}
  {\begin{tcolorbox}[enhanced, breakable, colback=code-bg, colframe=code-frame, boxrule=0.5pt, arc=3mm, shadow={0mm}{0mm}{0mm}{white}]}
  {\end{tcolorbox}}
\fi

% 4. Fix line wrapping
\DefineVerbatimEnvironment{Highlighting}{Verbatim}{
  breaklines=true,
  breakanywhere=true,
  commandchars=\\\{\}
}

\AtBeginDocument{
    \hypersetup{
      breaklinks=true,
      pdfborder={0 0 0}
    }
}

% Customized Lists
\usepackage{enumitem}

% 1. Set global settings for bullet lists (itemize) and numbered lists (enumerate)
%    nosep      = kills ALL vertical space (top, bottom, and between items)
%    itemsep=... = adds a tiny bit of breathing room between items (optional)
\setlist[itemize]{nosep, itemsep=0pt}
\setlist[enumerate]{nosep, itemsep=0pt}

% 2. Fix indentation (Optional but looks "modern")
%    This aligns the bullet with the left margin, rather than indented
\setlist[itemize,1]{leftmargin=*}
\setlist[enumerate,1]{leftmargin=*}
